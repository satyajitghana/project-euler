Largest Product In A Series#

Python Solution#

def calculate_product(part_of_num_string):
    product = 1
    for i in range(len(part_of_num_string) - 1):
        product *= int(part_of_num_string[i])

    return product

calc :: String -> Int
calc x = foldl (*) 1 $ map (\x -> read [x] :: Int) x

slice :: [a] -> Int -> Int -> [a]
slice x l t = take t $ drop l x
